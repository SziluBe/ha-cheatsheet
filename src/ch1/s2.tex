\wde{1.2 Conv. seq.} $(x_n) \to a \in \R$ iff $\forall \epsilon > 0, \exists N \in \N$
s.t. $n \ge N \Rightarrow |x_n - a| < \epsilon$.
\wde{1.3 Cauchy seq.} $(x_n)$ Cauchy iff $\forall \epsilon > 0, \exists N \in \N$
s.t. $m, n \ge N \Rightarrow |x_n - x_m| < \epsilon$.
\wt{1.4 Real conv. seq. iff Real Cauchy seq.}
\wde{1.4 Subsequence} $(x_{n_k})$ is a subsequence of $(x_n)$ iff $n_1 < n_2 < \dots$.
\wt{1.5 Bolzano-Weierstrass} Every bounded seq. of reals has a convergent subsequence.
\wde{1.5 Lim. inf., lim. sup.} $\liminf_{n \to \infty} x_n = \lim_{n \to \infty} \inf_{k \ge n} x_k$,
$\limsup_{n \to \infty} x_n = \lim_{n \to \infty} \sup_{k \ge n} x_k$.
\wt{1.6 Real seq. conv. iff lim. inf. = lim. sup.}
